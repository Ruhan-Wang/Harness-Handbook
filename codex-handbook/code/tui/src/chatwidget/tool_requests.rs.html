           self.add_boxed_history(cell);            self.request_redraw();            return;        }        if ev.status != GuardianAssessmentStatus::Denied {            return;        }        self.review.recent_auto_review_denials.push(ev.clone());        let cell = if let Some(command) = guardian_command(&ev.action) {            history_cell::new_approval_decision_cell(                history_cell::ApprovalDecisionSubject::Command(command),                crate::history_cell::ReviewDecision::Denied,                history_cell::ApprovalDecisionActor::Guardian,            )        } else {            match &ev.action {                GuardianAssessmentAction::ApplyPatch { files, .. } => {                    let files = files                        .iter()                        .map(|path| path.display().to_string())                        .collect::<Vec<_>>();
