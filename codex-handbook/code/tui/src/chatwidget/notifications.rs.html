Desktop notification coalescing for `ChatWidget`.use super::*;impl ChatWidget {    pub(super) fn notify(&mut self, notification: Notification) {        if !notification.allowed_for(&self.config.tui_notifications.notifications) {            return;        }        if let Some(existing) = self.pending_notification.as_ref()            && existing.priority() > notification.priority()        {            return;        }        self.pending_notification = Some(notification);        self.request_redraw();    }    pub(crate) fn maybe_post_pending_notification(&mut self, tui: &mut crate::tui::Tui) {        if let Some(notif) = self.pending_notification.take() {            tui.notify(notif.display());        }
